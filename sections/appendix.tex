\subsubsection{Encoding Algorithm}
\begin{algorithm}
\caption{Arithmetic Encoding}
\begin{algorithmic}
\STATE $L \leftarrow 0$, $H \leftarrow 1$
\STATE $pending \leftarrow 0$
\FOR{each symbol $s$ in input}
    \STATE $range \leftarrow H - L$
    \STATE $H \leftarrow L + range \times P_{high}(s)$
    \STATE $L \leftarrow L + range \times P_{low}(s)$
    \WHILE{$H < 0.5$ OR $L \geq 0.5$}
        \IF{$H < 0.5$}
            \STATE Output 0
            \STATE Output $pending$ ones
            \STATE $pending \leftarrow 0$
        \ELSIF{$L \geq 0.5$}
            \STATE Output 1
            \STATE Output $pending$ zeros
            \STATE $pending \leftarrow 0$
            \STATE $L \leftarrow L - 0.5$
            \STATE $H \leftarrow H - 0.5$
        \ENDIF
        \STATE $L \leftarrow 2L$
        \STATE $H \leftarrow 2H$
    \ENDWHILE
    \WHILE{$L \geq 0.25$ AND $H < 0.75$}
        \STATE $pending \leftarrow pending + 1$
        \STATE $L \leftarrow 2(L - 0.25)$
        \STATE $H \leftarrow 2(H - 0.25)$
    \ENDWHILE
    \STATE Update probability model with symbol $s$
\ENDFOR
\end{algorithmic}
\end{algorithm}
